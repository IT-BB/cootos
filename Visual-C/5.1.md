###函数
```cpp
#include <iostream>

using namespace std;

long func1(long x,long y)		//函数头
{                              //函数体
	long z = x+y;
	return z;		//return返回的类型为函数头定义的long func1的long
}

int main()
{
	long s=func1(20,30);	//调用函数
	cout<<s<<endl;
	return 0;
}
```
输出
```text
50
```

#include <iostream>

###给函数传递实参
```cpp
#include <iostream>

using namespace std;

long ISc(long a, long b)		//形参,实参的副本，不能修改实参
{
	return a + b;
}

void main()
{
	long s1 = ISc(10, 20);		//传递实参
	cout << s1<< endl;
}
```
###指针传递
```cpp
#include <iostream>

using namespace std;

long ISc(long* a)	//接收的是long s2的指针地址
{
	*a += 15;
	return *a;
}

void main()
{
	long s2 = 20;
	long* s1(&s2);
	cout << "s2的内存地址： " << &s2 << " 值： " << s2 << endl;
	cout << "指针的内存地址： " << s1 << " 值： " << *s1 << endl;
	ISc(s1);	//传递的是s2的指针地址
	cout << "s2的内存地址： " << &s2 << " 值： " << s2 << endl;
	cout << "指针的内存地址： " << s1 << " 值： " << *s1 << endl;
}
```
输出
```text
s2的内存地址： 0025FBA4 值： 20
指针的内存地址： 0025FBA4 值： 20
s2的内存地址： 0025FBA4 值： 35
指针的内存地址： 0025FBA4 值： 35
```
###给函数传递数组
```cpp
#include <iostream>

using namespace std;

void ISptr(long arr[],int count)
{
	for (int i = 0; i < count; i++)
		cout << "数组的值是: " << arr[i] << endl;
}

void main()
{
	long ptr[] = { 10, 20, 30, 50, 100, 200 };
	ISptr(ptr, (sizeof ptr) / (sizeof ptr[0]));
}
```
数组没有记录其大小，所以不能对作为实参传递给函数的数组使用基于范围的for循环
###传递数组时使用指针传递
```cpp
#include <iostream>

using namespace std;

void ISptr(long* arr,int count)
{
	long sum = 0;
	for (int i = 0; i < count; i++)
		sum += arr[i];
	cout << sum << endl;
}

void main()
{
	long ptr[] = { 10, 20, 30, 50, 100, 200 };
	ISptr(ptr, (sizeof ptr) / (sizeof ptr[0]));
}
```
###按引用传递
```cpp
#include <iostream>

using namespace std;

void ISa(long& s)
{
	s++;
}

void main()
{
	long a = 10;
	ISa(a);
	cout << a << endl;
}
```
