###函数
```cpp
#include <iostream>

using namespace std;

long func1(long x,long y)		//函数头
{                              //函数体
	long z = x+y;
	return z;		//return返回的类型为函数头定义的long func1的long
}

int main()
{
	long s=func1(20,30);	//调用函数
	cout<<s<<endl;
	return 0;
}
```
输出
```text
50
```
void无返回值的函数
```cpp
#include <iostream>

using namespace std;

void func()		//void无返回值
{
	cout<<"Hello world"<<endl;
}

void main()		//void无返回值
{
	func();
}
```
输出
```text
Hello world
```
使用函数
```cpp
#include <iostream>

using namespace std;

long func(long x,long y);	//函数原型
void hello();			//函数原型

void main()
{
	hello();
	cout<<func(10,20)<<endl;
}

long func(long x,long y)
{
	long z = x+y;
	return z;
}

void hello()
{
	cout<<"hello world"<<endl;
}
```
输出
```text
hello world
30
```
