###struct 结构

```cpp
#include <iostream>
#include <iomanip>

using namespace std;

struct Box
{
	char* color;
	int size;
};

int main()
{
	Box b1;		//初始化结构
	b1.color="red";		//赋值
	b1.size=20;			//赋值
	Box b2={"green",50};	//初始化结构并赋值
	cout<<"Box 1: "<<b1.color<<setw(4)<<b1.size<<endl<<"Box 2: "<<b2.color<<setw(4)<<b2.size<<endl;		//访问结构成员
	return 0;
}
```
