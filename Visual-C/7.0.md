###struct 结构

```cpp
#include <iostream>
#include <iomanip>

using namespace std;

struct Box
{
	char* color;
	int size;
};

int main()
{
	Box b1;
	b1.color="red";
	b1.size=20;
	Box b2={"green",50};
	cout<<"Box 1: "<<b1.color<<setw(4)<<b1.size<<endl<<"Box 2: "<<b2.color<<setw(4)<<b2.size<<endl;
	return 0;
}
```
