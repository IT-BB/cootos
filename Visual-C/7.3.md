###类的静态成员
```cpp
#include <iostream>

using namespace std;

class box
{
public:
	static int objectCount;		//类的静态数据成员

	explicit box(long s1,long s2=20,long s3=30)	//s1不能用值，有值就运行不了了
	{
	cout<<"explicit box "<<endl;
	m_Height=s2;
	m_Length=s1;
	m_Width=s3;
	objectCount++;
	}

	box()
	{
		cout<<"This is new box"<<endl;
		m_Width=m_Height=m_Length=10;
		objectCount++;
	}

	long boxs() const
	{
		return m_Length*m_Width*m_Height;
	}

private:
	long m_Width;
	long m_Height;
	long m_Length;
};

int box::objectCount=0;		//类外部进行静态数据成员的初始化

int main()
{
	box bxs[5];
	box b1(20,30,40);
	cout<<endl;
	cout<<"box[5] of objects = "<<bxs[3].objectCount<<endl;		//静态数据成员，要被同类的所有对象共享
	cout<<"box1 of objects = "<<b1.objectCount<<endl;
	return 0;
}
```
输出:
```text
This is new box
This is new box
This is new box
This is new box
This is new box
explicit box

box[5] of objects = 6
box1 of objects = 6
```
