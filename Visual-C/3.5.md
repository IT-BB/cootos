###for循环
依次输出1-100
```cpp
#include <iostream>

using namespace std;

int main()
{
	for(int i = 1;i<=100;i++)
	{
		cout<<i<<endl;
	}
	return 0;
}
```
输出累加1-100的和
```cpp
#include <iostream>

using namespace std;

int main()
{
	int sum=0;
	for(int i = 1;i<=100;i++)
	{
		sum+=i;
	}
	cout<<sum<<endl;
	return 0;
}
```
