指针数组
```cpp
#include <iostream>

using namespace std;

int main()
{
	char* pstr[]={"Robert Redford",
					"Hopalong Cassidy",
					"Lassie",
					"Slim Pickens",
					"Boris Karloff",
					"Oliver Hardy"
	};
	char* pstart("Your lucky star is ");
	for(int i=0;i<_countof(pstr);i++)
	{
		cout<<pstart<<pstr[i]<<endl;
	}
	return 0;
}
```
