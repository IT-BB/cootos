类函数
```cpp
#include <iostream>

using namespace std;

class Box
{
public:
    char* color;
    int size;
    long box(void); //函数原型
	long boxs()		//添加成员函数
	{
		return size*size;
	};
};

int main()
{
    Box b1;
    b1.color="red";
    b1.size=20;
    cout<<"Box b1 color is "<<b1.color<<" and size is "<<b1.size<<endl;
    cout<<b1.box()<<endl;
	cout<<b1.boxs()<<endl;
    return 0;
}

long Box::box()     //该函数属于Box类，即给函数加上类名作为前缀
{
    return ++size;
}
```
构造函数
```cpp
#include <iostream>

using namespace std;

class Box
{
public:
	char* color;
	long size;

	Box(char* colors,long sizes)		//构造函数,不能有任何返回类型
	{
		cout<<"This is Box color is "<<colors<<" and size is "<<sizes<<endl;
	};
};

int main()
{
	Box b1("red",25);
	return 0;
}
```
输出：
```text
This is Box color is red and size is 25
```
默认的构造函数
```cpp
#include <iostream>

using namespace std;

class Box
{
public:
    char* color;
    long size;

    Box(char* colors,long sizes)        //构造函数
    {
        cout<<"This is Box color is "<<colors<<" and size is "<<sizes<<endl;
		color=colors;		//必须添加，否则boxs()函数出错
		size=sizes;			//必须添加，否则boxs()函数出错
    }

    Box()           //默认的构造函数
    {
        cout<<"This is Default Box."<<endl;
    }

    long boxs()
    {
        return size*size;
    }
};

int main()
{
    Box b1("red",20);
    Box b2;
    cout<<b1.boxs()<<endl;
    return 0;
}
```
输出:
```text
This is Box color is red and size is 25
This is Default Box.
400
```
默认的形参值
```cpp
#include <iostream>

using namespace std;

class Box
{
public:
	char* color;
	long size;

	Box(char* colors="red",long sizes=50)		//带默认形参值的构造函数
	{
		color=colors;
		size=sizes;
		cout<<"This is Box color is "<<color<<" and size is "<<size<<endl;
	}
};

int main()
{
	Box b1;
	Box b2("green",20);
	return 0;
}
```
输出：
```text
This is Box color is red and size is 50
This is Box color is green and size is 20
```
在构造函数中使用初始化列表
```cpp
#include <iostream>

using namespace std;

class Box
{
public:
	char* color;
	long size;

	Box(char* colors="red",long sizes=50):			//在构造函数中使用初始化列表
		color(colors),size(sizes)
	{
		cout<<"This is Box color is "<<color<<" and size is "<<size<<endl;
	}
};

int main()
{
	Box b1;
	Box b2("green",20);
	Box b3("black",30);
	return 0;
}
```
输出：
```text
This is Box color is red and size is 50
This is Box color is green and size is 20
This is Box color is black and size is 30
```
