###while循环
输出1-100
```cpp
#include <iostream>

using namespace std;

int main()
{
	int i =1;
	const int max=100;

	while(i<=max)
	{
		cout<<i<<endl;
		i++;
	}
	return 0;
}
```
累加1-100的和
```cpp
#include <iostream>

using namespace std;

int main()
{
	int i =1,sum = 0;
	const int max=100;

	while(i<=max)
	{
		sum+=i;
		i++;
	}
	cout<<sum<<endl;
	return 0;
}
```
