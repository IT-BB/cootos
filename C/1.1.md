求两个数字中最大的值
```c
#include<stdio.h>
void main()			//主函数
{
	int max(int x,int y);	//对被调用函数max的声明
	int a,b,c;
	scanf("%d,%d",&a,&b);	//输入变量a和b的值
	c=max(a,b);				//调用max函数，将得到的值赋给c
	printf("max=%d\n", c);
}

int max(int x,int y)
{
	int z;
	if(x>y) z=x;
	else z=y;
	return(z);
}
```
运行后输入：
```bash
8,5
max=8
```
